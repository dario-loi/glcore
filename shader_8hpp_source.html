503 inline auto shader_program::parse_shaders(std::string_view source) const -> std::vector<shader>
504 {
505  std::vector<shader> shaders;
506  std::string_view const type_token { "#type" };
507 
508  std::size_t pos { source.find(type_token, 0) };
509  while (pos != std::string::npos) {
510  const std::size_t eol { source.find_first_of("\r\n", pos) };
511  const std::size_t begin { pos + type_token.size() + 1 };
512  const std::size_t next_line_pos { source.find_first_not_of("\r\n", eol) };
513  std::string const type { source.substr(begin, eol - begin) };
514 
515  auto const shader_type { string_to_shader_type(type) };
516 
517  // With C++23 we could drastically simplify this thanks to std::optional's monadic operations.
518  if (!shader_type.has_value()) [[unlikely]] {
519 #ifdef STAPLEGL_DEBUG
520  std::fprintf(stderr, STAPLEGL_LINEINFO ", invalid shader type \"%s\"\n",
521  type.data());
522 #endif // STAPLEGL_DEBUG
523  return {};
524  }
525 
526  pos = source.find(type_token, next_line_pos);
527  shaders.push_back({ shader_type.value(),
528  (pos == std::string::npos)
529  ? std::string(source.substr(next_line_pos))
530  : std::string(source.substr(next_line_pos, pos - next_line_pos)) });
531  }
