518 {
519  std::vector<shader> shaders;
520  std::string_view const type_token { "#type" };
521 
522  std::size_t pos { source.find(type_token, 0) };
523  while (pos != std::string::npos) {
524  const std::size_t eol { source.find_first_of("\r\n", pos) };
525  const std::size_t begin { pos + type_token.size() + 1 };
526  const std::size_t next_line_pos { source.find_first_not_of("\r\n", eol) };
527  std::string const type { source.substr(begin, eol - begin) };
528 
529  auto const shader_type { string_to_shader_type(type) };
530 
531  // With C++23 we could drastically simplify this thanks to std::optional's monadic operations.
532  if (!shader_type.has_value()) [[unlikely]] {
533 
534 #ifdef STAPLEGL_DEBUG
535  std::fprintf(stderr, STAPLEGL_LINEINFO ", invalid shader type \"%s\"\n",
536  type.data());
537 #endif // STAPLEGL_DEBUG
538  return {};
539  }
540 
541  pos = source.find(type_token, next_line_pos);
542  shaders.push_back({ shader_type.value(),
543  (pos == std::string::npos)
